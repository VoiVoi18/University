#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <locale.h>
int main() {
    int i,maxim,minim,N;
    int* a;  
    system("chcp 1251");
    printf("Введите размер массива: ");
    scanf("%d", &N);
    a = (int*)malloc(N * sizeof(int));
    srand(time(NULL));
    for (int i = 0; i < N; i++)
    {
        a[i] = rand() % 1000;
        printf("%d ", a[i]);
    }
    maxim = a[0];
    minim = a[N-1];
    for (int i = 0; i < N; i++)
    {
        if (a[i] > maxim) {
            maxim = a[i];
        }
    }
    for (int i = 0; i < N; i++)
    {
        if (a[i] < minim) {
            minim = a[i];
        }
    }
    printf("\n");
    printf("Максимальный элемент: %d", maxim);
    printf("\n");
    printf("Минимальный элемент: %d", minim);
    printf("\n");
    printf("Разница между максимальным и минимальным элементом: %d", maxim - minim);
    getchar();
    return 0;
}
